5   public class CommandLineParser {
6     private final String name;
7     private final String gender;
8     private final String gpa;
9     private final ArrayList<String> classes = new ArrayList<>();
10  
11    public CommandLineParser(String... args) {
12      if (args.length < 3) {
13        throw new Student.MissingCommandLineArgumentsException();
14      }
15  
16      this.name = args[0];
17      this.gender = args[1];
18      this.gpa = args[2];
19  
20      for (int i = 3; i < args.length; i++) {
21        this.classes.add(args[i]);
22      }
23    }
24  
25    public Student createStudent() {
26      return new Student(name, classes, parseGpa(), gender);
27    }
28  
29    protected double parseGpa() {
30      double gpa;
31      try {
32        gpa = Double.parseDouble(this.gpa);
33  
34      } catch (NumberFormatException ex) {
35        throw new Student.InvalidGpaException(this.gpa);
36      }
37      return gpa;
38    }
