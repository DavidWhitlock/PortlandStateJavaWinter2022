15  /**
16   * Unit tests for the Student class.  In addition to the JUnit annotations,
17   * they also make use of the <a href="http://hamcrest.org/JavaHamcrest/">hamcrest</a>

20  public class StudentTest

22  
23    @Test
24    void studentNamedPatIsNamedPat() {
25      String name = "Pat";
26      var pat = createStudentNamed(name);
27      assertThat(pat.getName(), equalTo(name));
28    }
29  
30    @Test
31    void nameLongerThan40CharactersThrowsIllegalArgumentException() {
32      String tooLongName = "12345678901234567890123456789012345678901";
33      assertThrows(IllegalArgumentException.class, () -> createStudentNamed(tooLongName));
34    }
35  
36    private Student createStudentNamed(String studentName) {
37      return new Student(studentName, new ArrayList<>(), 0.0, "other");
38    }
39  
40    @Test
41    void femaleGenderHasShePronoun() {
42      Student femaleStudent = createStudentWithGender(Student.FEMALE_GENDER);
43      assertThat(femaleStudent.toString(), containsString(Student.FEMALE_PRONOUN));
44    }
45  
46    private Student createStudentWithGender(String gender) {
47      return new Student("Name", new ArrayList<>(), 0.0, gender);
48    }
49  
50    @Test
51    void maleGenderHasHePronoun() {
52      Student maleStudent = createStudentWithGender(Student.MALE_GENDER);
53      assertThat(maleStudent.toString(), containsString(Student.MALE_PRONOUN));
54    }
55  
56    @Test
57    void otherGenderHasTheyPronoun() {
58      Student maleStudent = createStudentWithGender(Student.OTHER_GENDER);
59      assertThat(maleStudent.toString(), containsString(Student.THEY_PRONOUN));
60    }
61  
62    @Test
63    void otherGenderCaseInsensitiveHasTheyPronoun() {
64      Student maleStudent = createStudentWithGender("OTHER");
65      assertThat(maleStudent.toString(), containsString(Student.THEY_PRONOUN));
66    }
67  
68    @Test
69    void unrecognizedGenderThrowsUnrecognizedGenderException() {
70      String unrecognizedGender = "unsupported";
71      UnrecognizedGenderException ex =
72        assertThrows(UnrecognizedGenderException.class, () -> createStudentWithGender(unrecognizedGender));
73      assertThat(ex.getMessage(), containsString(unrecognizedGender));
74    }
75  
76    @Test
77    void toStringContainsStudentName() {
78      String studentName = "Dave";

81    }
82  
83    @Test
84    void tooLargeGpaThrowAnInvalidGpaException() {
85      double invalidGpa = 4.01;
86      InvalidGpaException ex =
87        assertThrows(InvalidGpaException.class, () -> createStudentWithGpa(invalidGpa));
88      assertThat(ex.getMessage(), containsString(String.valueOf(invalidGpa)));
89  
90    }

109   @Test
110   void toStringContainsEnrolledClasses() {
111     String class1 = "Operating Systems";
112     String class2 = "Compilers";
113     String class3 = "Algorithms";
114 
115     ArrayList<String> classes = new ArrayList<>();
116     classes.add(class1);
117     classes.add(class2);
118     classes.add(class3);
119 
120     Student student = new Student("Name", classes, 3.45, "other");
121     String toString = student.toString();
122 
123     assertThat(toString, StringContains.containsString(class1));
124     assertThat(toString, StringContains.containsString(class2));
125     assertThat(toString, StringContains.containsString(class3));
