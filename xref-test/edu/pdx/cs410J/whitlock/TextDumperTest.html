13  public class TextDumperTest {
14  
15    @Test
16    void airlineNameIsDumpedInTextFormat() {
17      String airlineName = "Test Airline";
18      Airline airline = new Airline(airlineName);
19  
20      StringWriter sw = new StringWriter();
21      TextDumper dumper = new TextDumper(sw);
22      dumper.dump(airline);
23  
24      String text = sw.toString();
25      assertThat(text, containsString(airlineName));
26    }
27  
28    @Test
29    void canParseTextWrittenByTextDumper(@TempDir File tempDir) throws IOException, ParserException {
30      String airlineName = "Test Airline";
31      Airline airline = new Airline(airlineName);
32  
33      File textFile = new File(tempDir, "airline.txt");
34      TextDumper dumper = new TextDumper(new FileWriter(textFile));
35      dumper.dump(airline);
36  
37      TextParser parser = new TextParser(new FileReader(textFile));
38      Airline read = parser.parse();
39      assertThat(read.getName(), equalTo(airlineName));
40    }
