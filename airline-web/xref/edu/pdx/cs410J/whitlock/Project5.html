7   /**
8    * The main class that parses the command line and communicates with the
9    * Airline server using REST.
10   */
11  public class Project5 {
12  
13      public static final String MISSING_ARGS = "Missing command line arguments";
14  
15      public static void main(String... args) throws IOException {
16          String hostName = null;
17          String portString = null;
18          String airlineName = null;
19          String flightNumberString = null;
20  
21          for (String arg : args) {
22              if (arg.equals("-README")) {
23                  printReadmeAndExit();
24  
25              } else if (hostName == null) {
26                  hostName = arg;
27  
28              } else if ( portString == null) {
29                  portString = arg;
30  
31              } else if (airlineName == null) {
32                  airlineName = arg;
33  
34              } else if (flightNumberString == null) {
35                  flightNumberString = arg;
36  
37              } else {
38                  usage("Extraneous command line argument: " + arg);
39              }
40          }
41  
42          if (hostName == null) {
43              usage( MISSING_ARGS );
44              return;
45  
46          } else if ( portString == null) {
47              usage("Missing port");
48              return;
49  
50          } else if (airlineName == null) {
51              usage("Missing airline name");
52              return;
53          }
54  
55          int port;
56          try {
57              port = Integer.parseInt( portString );
58  
59          } catch (NumberFormatException ex) {
60              usage("Port \"" + portString + "\" must be an integer");
61              return;
62          }
63  
64          AirlineRestClient client = new AirlineRestClient(hostName, port);
65  
66          try {
67              if (flightNumberString == null) {
68                  // Pretty print airline
69                  Airline airline = client.getAirline(airlineName);
70                  new PrettyPrinter(new OutputStreamWriter(System.out)).dump(airline);
71  
72              } else {
73                  // Add the flight
74                  client.addFlight(airlineName, new Flight(Integer.parseInt(flightNumberString)));
75              }
76  
77          } catch (IOException | ParserException ex ) {
78              error("While contacting server: " + ex);
79              return;
80          }
81  
82          System.exit(0);
83      }
84  
85      private static void printReadmeAndExit() throws IOException {
86          InputStream readme = Project5.class.getResourceAsStream("README.txt");
87          try (
88            BufferedReader reader = new BufferedReader(new InputStreamReader(readme))
89            ) {
90              for (String line = reader.readLine(); line != null; line = reader.readLine()) {
91                  System.out.println(line);
92              }
93          }
94  
95      }
96  
97      private static void error( String message )
98      {
99          PrintStream err = System.err;
100         err.println("** " + message);
101 
102         System.exit(1);
103     }
104 
105     /**
106      * Prints usage information for this program and exits
107      * @param message An error message to print
108      */
109     private static void usage( String message )
110     {
111         PrintStream err = System.err;
112         err.println("** " + message);
113         err.println();
114         err.println("usage: java Project5 host port [word] [definition]");
115         err.println("  host         Host of web server");
116         err.println("  port         Port of web server");
117         err.println("  word         Word in dictionary");
118         err.println("  definition   Definition of word");
119         err.println();
120         err.println("This simple program posts words and their definitions");
121         err.println("to the server.");
122         err.println("If no definition is specified, then the word's definition");
123         err.println("is printed.");
124         err.println("If no word is specified, all dictionary entries are printed");
125         err.println();
126 
127         System.exit(1);
128     }
